Class: ReferralService

    (bot: Pyrogram Client, used for channel membership checks)

Methods (ALL async):

    async def register_referral(self, referrer_id: int, referred_id: int) -> bool
        Validation checks (return False immediately if any fail):
        1. referrer_id == referred_id → return False (self-referral)
        2. Check if referred_id already has a referral in DB → return False
        3. Create pending referral via repository.create_pending()
        4. Ensure referrer wallet exists via repository.upsert_wallet()
        Return True on success.
        Log all rejection reasons with structlog including referrer_id, referred_id, reason.

    async def check_membership(self, user_id: int, channel_id: int) -> bool
        Call bot.get_chat_member(channel_id, user_id).
        Return True if status is member/administrator/creator.
        Return False if ChatMemberStatus is BANNED, LEFT, or RESTRICTED.
        Wrap in try/except UserNotParticipant → return False.
        Wrap in try/except FloodWait as e → await asyncio.sleep(e.value) then retry once.

    async def qualify_pending_referrals(self, channel_id: int) -> int
        Called by background job.
        Get all pending referrals older than 24 hours via repository.get_pending_older_than(24).
        For each:
            - Check channel membership via check_membership()
            - If member: repository.qualify_referral(), repository.increment_balance(referrer_id, +1)
            - If not member: skip (leave pending, TTL will clean it)
        Return count of qualified referrals.
        Use asyncio.sleep(0.1) between each check to avoid burst API calls.

    async def handle_member_left(self, user_id: int) -> None
        Called when a user leaves the required channel.
        Get referral where referred_user_id=user_id and status=QUALIFIED.
        If found: repository.invalidate_referral(user_id), repository.decrement_balance(referrer_id).
        Log with structlog.

    async def handle_member_rejoined(self, user_id: int, channel_id: int) -> None
        Called when a user rejoins the required channel.
        Get referral where referred_user_id=user_id and status=INVALIDATED.
        If found: verify channel membership via check_membership().
        If confirmed member: repository.reactivate_referral(user_id), repository.increment_balance(referrer_id, +1).

    async def get_wallet(self, user_id: int) -> Optional[dict]
        Return wallet dict or None via repository.get_wallet().

    async def snapshot_discount(self, user_id: int, points_to_use: int, plan_price: int) -> dict
        IMPORTANT: This locks a discount snapshot. Call this at payment creation, never at checkout display.
        Validate: points_to_use <= wallet.points_balance
        Validate: points_to_use <= plan_price (cannot discount more than plan price)
        Deduct via repository.deduct_points(user_id, points_to_use).
        If deduction failed (insufficient balance): raise ValueError('Insufficient referral balance')
        Return: {'original_price': plan_price, 'discount_applied': points_to_use, 'final_price': plan_price - points_to_use, 'snapshotted_at': datetime.utcnow().isoformat()}
